}

void Sort(struct Student** group, int n, CompareTwoStudentsFunc compare)
{
	struct Student *temp;
	for (int j = 1; j < n; j++){
		for (int i = 0; i < n - j; i++){
			if (compare(group[i], group[i + 1]) == 1){
				temp = group[i];
				group[i] = group[i + 1];
				group[i + 1] = temp;
			}
		}
	}
}


int CountIf(Student** group, int n, StudentCompareFunc compare)
{
	int count = 0;
	for (int i = 0; i < n; i++) {
		if (compare(group[i])) {
			count++;
		}
	}
	return count;
}
